import subprocess, sys, yaml

def frontmatter(text):
    lines = text.splitlines()
    if not lines or lines[0].strip() != '---':
        return None
    end = None
    for i, l in enumerate(lines[1:], start=1):
        if l.strip() == '---':
            end = i
            break
    if end is None:
        return None
    return yaml.safe_load(chr(10).join(lines[1:end])) or {}

path = 'methodology/02_adversarial_tdd.md'
cur_text = open(path, encoding='utf-8').read()
cur_fm = frontmatter(cur_text)
if cur_fm is None:
    print(path + ': could not parse current frontmatter')
    sys.exit(1)
version = cur_fm.get('version')
cur_entries = cur_fm.get('change_log') or []

if version not in versions:
    print('no change_log entry matching current version ' + repr(version))
