import yaml, sys
f = 'docs/architecture/ADR-002-html-user-guide.md'
lines = open(f).read().split(chr(10))
assert lines[0] == '---', 'first line not ---'
end = None
for i, l in enumerate(lines[1:], start=1):
    if l == '---':
        end = i
        break
assert end is not None, 'no closing ---'
block = chr(10).join(lines[1:end])
fm = yaml.safe_load(block)
assert isinstance(fm, dict), 'frontmatter did not parse to mapping'
src = fm.get('authoritative_source', '')

import re, subprocess
prev = subprocess.run(['git','show','HEAD:.agent/product_spec.md'], capture_output=True, text=True, check=True).stdout
cur = open('.agent/product_spec.md').read()
def extract_entry(text, version):
    m = re.search(r'  - version: \"'+re.escape(version)+r'\"\n    date: \"[^\"]*\"\n    summary: \"[^\"]*\"', text)
    return m.group(0) if m else None
ok = True
for v in ['1.0.0','1.1.0']:
    p = extract_entry(prev, v)
    c = extract_entry(cur, v)
    same = (p is not None and p == c)
    print(v, 'byte-for-byte preserved:', same)
    if not same:
        ok = False
        print('  PREV:', repr(p))
        print('  CUR :', repr(c))
if not ok:
    raise SystemExit(1)
print('OK: both prior change_log entries preserved byte-for-byte')
"

      "$pfx"*) ok=1 ;;
    esac
  done
  if [ "$ok" -eq 0 ]; then
    echo "UNEXPECTED CHANGED FILE: $cf"
    unexpected=1
  fi

  echo "OK: no edits outside the permitted file set"

  fail=1

  echo "OK: validate.sh exited 0 with exact summary line 'SUMMARY: 8 passed, 0 failed'"
else
  echo "FAIL: validate.sh rc=$rc summary_ok=$summary_ok"
  fail=1
fi

echo
if [ "$fail" -eq 0 ]; then
  echo "ADVERSARIAL RESULT: ALL PROBES PASSED"

  echo "ADVERSARIAL RESULT: ONE OR MORE PROBES FAILED"
