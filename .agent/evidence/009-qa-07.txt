(script: v07.sh)
files=(CONTEXT.md methodology/00_planning.md methodology/01_execution.md methodology/02_adversarial_tdd.md methodology/03_circuit_breaker.md methodology/04_tool_driven_state.md methodology/05_release_train.md methodology/README.md registry/README.md registry/scope_definition_patterns.md schema/README.md standard/AGF.md standard/anti_hallucination.md standard/GIP.md standard/NDS.md standard/README.md templates/ADR_TEMPLATE.md templates/AGENTS.md templates/CONTEXT.md templates/DEBT.md templates/README.md tools/interface.md tools/README.md tools/SKILL.md)
fail=0
for f in "${files[@]}"; do
  test -f "$f" || { echo "MISSING $f"; fail=1; continue; }
  stale=$(grep -c 'authoritative_source: nizam-framework/' "$f" || true)
  if [ "$stale" -ne 0 ]; then echo "STALE PREFIX in $f"; fail=1; fi
  actual=$(grep '^authoritative_source:' "$f" | head -1 | sed 's/^authoritative_source: *//')
  if [ "$actual" != "$f" ]; then echo "MISMATCH in $f: authoritative_source='$actual' expected='$f'"; fail=1; fi
done
test "$fail" -eq 0 && echo PASS || { echo 'authoritative_source regression found above'; exit 1; }
PASS
