import json, os
feature_list = json.load(open(".agent/feature_list_002.json"))
features = {f["id"]: f for f in feature_list["features"]}
all_ok = True
for fid in ["011", "012", "013", "014", "015"]:
    print("--- feature " + fid + " ---")
    contract = json.load(open(".agent/contracts/" + fid + ".json"))
    status = contract.get("status")
    approvals = contract.get("approvals", {})
    print("contract status: " + str(status) + " approvals: " + str(approvals))
    contract_ok = status in ("implemented", "complete") and approvals.get("validator_mode_a") and approvals.get("evaluator_contract_review")
    qa_path = ".agent/qa/" + fid + ".json"
    if not os.path.isfile(qa_path):
        print("qa verdict not yet on disk (expected only for 015, in-flight in this QA round)")
        qa_ok = (fid == "015")
    else:
        qa = json.load(open(qa_path))
        print("qa verdict: " + str(qa.get("verdict")))
        qa_ok = (qa.get("verdict") == "pass")
    verify_files = sorted([p for p in os.listdir(".agent/evidence") if p.startswith(fid + "-verify-")])
    qa_files = sorted([p for p in os.listdir(".agent/evidence") if p.startswith(fid + "-qa-")])
    print(fid + "-verify-* count: " + str(len(verify_files)) + "  " + fid + "-qa-* count: " + str(len(qa_files)))
    evidence_ok = len(verify_files) > 0 and len(qa_files) > 0
    fl_status = features.get(fid, {}).get("status")
    print("feature_list_002.json status: " + str(fl_status))
    fl_ok = (fl_status == "complete") if fid != "015" else (fl_status == "pending")
    ok = contract_ok and qa_ok and evidence_ok and fl_ok
    print("OVERALL " + fid + ": " + ("OK" if ok else "FAIL"))
    all_ok = all_ok and ok
print("SUMMARY: " + ("ALL_OK (015 correctly in-flight)" if all_ok else "FAIL"))
