import subprocess, sys, yaml

def frontmatter(text):
    lines = text.splitlines()
    if not lines or lines[0].strip() != '---':
        return None
    end = None

        if l.strip() == '---':
            end = i
            break
    if end is None:
        return None
    return yaml.safe_load(chr(10).join(lines[1:end])) or {}
