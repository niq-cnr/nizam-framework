            ok = True
            break
        except jsonschema.ValidationError:
            pass
    if not ok:
        bad.append(f)
if bad:
    print('FAILED to validate against either schema: ' + ', '.join(bad))
    sys.exit(1)
print(f'PASS ({len(files)} qa artifacts, all validate)')
